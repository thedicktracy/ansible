- hosts:
    - host_list
  vars_prompt:
    - name: root_passwd
      prompt: "Enter new root password "
      private: yes
      encrypt: sha512_crypt
      confirm: true
      salt_size: 7
  tasks:
    - name: Change root passwd
      shell: echo -n '{{ root_passwd }}' | passwd root --stdin
      register: results
      become: true
      become_user: root
    - name: Printing the results
      debug:
        msg: "{{results.stdout_lines}}"
